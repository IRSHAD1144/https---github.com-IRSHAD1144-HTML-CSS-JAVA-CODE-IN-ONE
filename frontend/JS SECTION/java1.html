🔄 CONDITIONALS & LOOPS

if | else | else if | switch | ternary

Conditional Statements

// if-else statement
let score = 85;
if (score >= 90) {
    console.log("Grade A");
} else if (score >= 80) {
    console.log("Grade B");
} else if (score >= 70) {
    console.log("Grade C");
} else {
    console.log("Grade F");
}

// switch statement
let day = "Monday";
switch(day) {
    case "Monday":
        console.log("Start of week");
        break;
    case "Friday":
        console.log("Weekend coming");
        break;
    default:
        console.log("Mid week");
}

// Ternary operator
let age = 20;
let status = age >= 18 ? "Adult" : "Minor";

for | while | do-while | for-in | for-of

Loops

// for loop
for (let i = 0; i < 5; i++) {
    console.log(i);  // 0,1,2,3,4
}

// while loop
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

// do-while loop (runs at least once)
let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

// for-in loop (objects)
let person = {name: "John", age: 30};
for (let key in person) {
    console.log(key + ": " + person[key]);
}

// for-of loop (arrays)
let colors = ["red", "green", "blue"];
for (let color of colors) {
    console.log(color);
}

📊 ARRAY METHODS (35+)

push | pop | shift | unshift

Array Modification Methods

let arr = [1, 2, 3];

arr.push(4);     // [1, 2, 3, 4] - add to end
arr.pop();       // [1, 2, 3] - remove from end
arr.unshift(0);  // [0, 1, 2, 3] - add to beginning
arr.shift();     // [1, 2, 3] - remove from beginning

[1, 2, 3]

map | filter | reduce | forEach

Array Iteration Methods

let numbers = [1, 2, 3, 4, 5];

// map - transform each element
let doubled = numbers.map(n => n * 2);  // [2, 4, 6, 8, 10]

// filter - keep elements that pass test
let evens = numbers.filter(n => n % 2 === 0);  // [2, 4]

// reduce - accumulate values
let sum = numbers.reduce((acc, curr) => acc + curr, 0);  // 15

// forEach - execute for each element
numbers.forEach(n => console.log(n));

// find - find first matching element
let found = numbers.find(n => n > 3);  // 4

// some - check if ANY element passes
let hasEven = numbers.some(n => n % 2 === 0);  // true

// includes - check if value exists
let hasThree = numbers.includes(3);  // true

[1, 2, 3, 4, 5]

class | constructor | extends | super

Classes (OOP)

// Class definition
class Animal {
    constructor(name) {
        this.name = name;
    }
    
    speak() {
        return this.name + " makes a sound";
    }
}

// Inheritance
class Dog extends Animal {
    constructor(name, breed) {
        super(name);
        this.breed = breed;
    }
    
    speak() {
        return this.name + " barks!";
    }
}

try | catch | throw | finally

Error Handling

try {
    // Code that might throw error
    let result = riskyOperation();
} catch (error) {
    // Handle error
    console.log(error.message);
} finally {
    // Always executes
    console.log("Cleanup");
}

// Throwing custom errors
